033    public static final String CRYPTO_RSA_PRIVATEKEY_FILEPATH = "crypto.rsa.privatekey.filepath";
034    public static final String CRYPTO_RSA_PUBLICKEY_FILEPATH = "crypto.rsa.publickey.filepath";

038    static final byte[] MAGIC_BYTES = new byte[] { (byte) 0xDF, (byte) 0xBB };
039    protected static final String DEFAULT_TRANSFORMATION = "AES/CBC/PKCS5Padding"; //TODO allow other like GCM
040    private static final Map<String, byte[]> aesKeyCache = new HashMap<String, byte[]>();
041    private static final int MAGIC_BYTES_LENGTH = MAGIC_BYTES.length;
042    private static final int HEADER_LENGTH = MAGIC_BYTES_LENGTH + 3;
043    private static final String AES = "AES";
044    private static final String RSA = "RSA";
045    private static final int RSA_MULTIPLICATOR = 128;
046    private int opMode;

067    //not thread safe
068    private class ConsumerCryptoBundle {
069
070        private Cipher rsaDecrypt;
071        final Cipher aesDecrypt = Cipher.getInstance(DEFAULT_TRANSFORMATION);
072
073        private ConsumerCryptoBundle(PrivateKey privateKey) throws Exception {
074            rsaDecrypt = Cipher.getInstance(RSA);
075            rsaDecrypt.init(Cipher.DECRYPT_MODE, privateKey);
076        }
077
078        private byte[] aesDecrypt(byte[] encrypted) throws KafkaException {
079            try {
080                if (encrypted[0] == MAGIC_BYTES[0] && encrypted[1] == MAGIC_BYTES[1]) {
081                    final byte hashLen = encrypted[2];
082                    final byte rsaFactor = encrypted[3];
083                    final byte ivLen = encrypted[4];
084                    final int offset = HEADER_LENGTH + hashLen + (rsaFactor * RSA_MULTIPLICATOR) + ivLen;
085                    final String aesHash = DatatypeConverter.printHexBinary(Arrays.copyOfRange(encrypted, HEADER_LENGTH, HEADER_LENGTH + hashLen));
086                    final byte[] iv = Arrays.copyOfRange(encrypted, HEADER_LENGTH + hashLen + (rsaFactor * RSA_MULTIPLICATOR),
087                            HEADER_LENGTH + hashLen + (rsaFactor * RSA_MULTIPLICATOR) + ivLen);
088
089                    byte[] aesKey;
090
091                    if ((aesKey = aesKeyCache.get(aesHash)) != null) {
092                        aesDecrypt.init(Cipher.DECRYPT_MODE, createAESSecretKey(aesKey), new IvParameterSpec(iv));
093                        return crypt(aesDecrypt, encrypted, offset, encrypted.length - offset);
094                    } else {
095                        byte[] rsaEncryptedAesKey = Arrays.copyOfRange(encrypted, HEADER_LENGTH + hashLen,
096                                HEADER_LENGTH + hashLen + (rsaFactor * RSA_MULTIPLICATOR));
097                        aesKey = crypt(rsaDecrypt, rsaEncryptedAesKey);
098                        aesDecrypt.init(Cipher.DECRYPT_MODE, createAESSecretKey(aesKey), new IvParameterSpec(iv));
099                        aesKeyCache.put(aesHash, aesKey);
100                        return crypt(aesDecrypt, encrypted, offset, encrypted.length - offset);
101                    }
102                } else {
103                    return encrypted; //not encrypted, just bypass decryption
104                }
105            } catch (Exception e) {
106                if(ignoreDecryptFailures) {
107                    return encrypted; //Probably not encrypted, just bypass decryption
108                }
109                
110                throw new KafkaException("Decrypt failed",e);
111            }
112        }
113    }

115    private class ThreadAwareKeyInfo {
116        private final SecretKey aesKey;
117        private final byte[] aesHash;
118        private final byte[] rsaEncyptedAesKey;
119        private final Cipher rsaCipher;
120        private final Cipher aesCipher;
121
122        protected ThreadAwareKeyInfo(PublicKey publicKey) throws Exception {
123            byte[] aesKeyBytes = new byte[aesKeyLen/8];
124            random.nextBytes(aesKeyBytes);
125            aesCipher = Cipher.getInstance(DEFAULT_TRANSFORMATION);
126            aesKey = createAESSecretKey(aesKeyBytes);
127            aesHash = hash(aesKeyBytes);
128            rsaCipher = Cipher.getInstance(RSA);
129            rsaCipher.init(Cipher.ENCRYPT_MODE, publicKey);
130            rsaEncyptedAesKey = crypt(rsaCipher, aesKeyBytes);
131        }
132    }

134    //threads safe
135    private class ProducerCryptoBundle {
136
137        private ThreadLocal<ThreadAwareKeyInfo> keyInfo = new ThreadLocal<ThreadAwareKeyInfo>() {
138            @Override
139            protected ThreadAwareKeyInfo initialValue() {
140                try {
141                    return new ThreadAwareKeyInfo(publicKey);
142                } catch (Exception e) {
143                    throw new KafkaException(e);
144                }
145            }
146        };
147        private final PublicKey publicKey;
148
149        private ProducerCryptoBundle(PublicKey publicKey) throws Exception {
150            this.publicKey = publicKey;
151        }
152
153        private void newKey() throws Exception {
154            final ThreadAwareKeyInfo ki = new ThreadAwareKeyInfo(publicKey);
155            keyInfo.set(ki);
156        }
157
158        private byte[] aesEncrypt(byte[] plain) throws KafkaException {
159            final ThreadAwareKeyInfo ki = keyInfo.get();
160
161            try {
162                final byte[] aesIv = new byte[16];
163                random.nextBytes(aesIv);
164                ki.aesCipher.init(Cipher.ENCRYPT_MODE, ki.aesKey, new IvParameterSpec(aesIv));
165                final byte[] prolog = concatenate(ki.aesHash, ki.rsaEncyptedAesKey, aesIv);
166                final byte[] header = concatenate(MAGIC_BYTES, new byte[] { (byte) ki.aesHash.length,
167                        (byte) (ki.rsaEncyptedAesKey.length / RSA_MULTIPLICATOR), (byte) aesIv.length });
168                return concatenate(header, prolog, crypt(ki.aesCipher, plain));
169            } catch (Exception e) {
170                throw new KafkaException(e);
171            }
172        }
173    }

175    protected void init(int opMode, Map<String, ?> configs, boolean isKey) throws KafkaException {
176        this.opMode = opMode;
177
178        final String hashMethodProperty = (String) configs.get(CRYPTO_HASH_METHOD);
179        
180        if(hashMethodProperty != null && hashMethodProperty.length() != 0) {
181            hashMethod = hashMethodProperty;
182        }
183        
184        final String ignoreDecryptFailuresProperty = (String) configs.get(CRYPTO_IGNORE_DECRYPT_FAILURES);
185        
186        if(ignoreDecryptFailuresProperty != null && ignoreDecryptFailuresProperty.length() != 0) {
187            ignoreDecryptFailures = Boolean.parseBoolean(ignoreDecryptFailuresProperty);
188        }
189        
190        final String aesKeyLenProperty = (String) configs.get(CRYPTO_AES_KEY_LEN);
191        
192        if(aesKeyLenProperty != null && aesKeyLenProperty.length() != 0) {
193            aesKeyLen = Integer.parseInt(aesKeyLenProperty);
194            if(aesKeyLen < 128 || aesKeyLen % 8 != 0) {
195                throw new KafkaException("Invalid aes key size, should be 128, 192 or 256");
196            }
197        }
198        
199        try {
200            if (opMode == Cipher.DECRYPT_MODE) {
201                //Consumer
202                String rsaPrivateKeyFile = (String) configs.get(CRYPTO_RSA_PRIVATEKEY_FILEPATH);
203                consumerCryptoBundle = new ConsumerCryptoBundle(createRSAPrivateKey(readBytesFromFile(rsaPrivateKeyFile)));
204            } else {
205                //Producer
206                String rsaPublicKeyFile = (String) configs.get(CRYPTO_RSA_PUBLICKEY_FILEPATH);
207                producerCryptoBundle = new ProducerCryptoBundle(createRSAPublicKey(readBytesFromFile(rsaPublicKeyFile)));
208            }
209        } catch (Exception e) {
210            throw new KafkaException(e);
211        }
212    }
213
214    protected byte[] crypt(byte[] array) throws KafkaException {
215        if (array == null || array.length == 0) {
216            return array;
217        }
218
219        if (opMode == Cipher.DECRYPT_MODE) {
220            //Consumer
221            return consumerCryptoBundle.aesDecrypt(array);
222        } else {
223            //Producer
224            return producerCryptoBundle.aesEncrypt(array);
225        }
226    }
227
228    /**
229     * Generate new AES key for the current thread
230     */
231    protected void newKey() {
232        try {
233            producerCryptoBundle.newKey();
234        } catch (Exception e) {
235            throw new KafkaException(e);
236        }
237    }

241    @SuppressWarnings("unchecked")
242    protected <T> T newInstance(Map<String, ?> map, String key, Class<T> klass) throws KafkaException {
243        Object val = map.get(key);
244        if (val == null) {
245            throw new KafkaException("No value for '" + key + "' found");
246        } else if (val instanceof String) {
247            try {
248                return (T) Utils.newInstance(Class.forName((String) val));
249            } catch (Exception e) {
250                throw new KafkaException(e);
251            }
252        } else if (val instanceof Class) {
253            return (T) Utils.newInstance((Class<T>) val);
254        } else {
255            throw new KafkaException("Unexpected type '" + val.getClass() + "' for '" + key + "'");
256        }
257    }

259    private static PrivateKey createRSAPrivateKey(byte[] encodedKey) throws NoSuchAlgorithmException, InvalidKeySpecException {
260        if (encodedKey == null || encodedKey.length == 0) {
261            throw new IllegalArgumentException("Key bytes must not be null or empty");
262        }
263
264        PKCS8EncodedKeySpec spec = new PKCS8EncodedKeySpec(encodedKey);
265        KeyFactory kf = KeyFactory.getInstance(RSA);
266        return kf.generatePrivate(spec);
267    }
268
269    private static SecretKey createAESSecretKey(byte[] encodedKey) {
270        if (encodedKey == null || encodedKey.length == 0) {
271            throw new IllegalArgumentException("Key bytes must not be null or empty");
272        }
273
274        return new SecretKeySpec(encodedKey, AES);
275    }
276
277    private static PublicKey createRSAPublicKey(byte[] encodedKey) throws NoSuchAlgorithmException, InvalidKeySpecException {
278        if (encodedKey == null || encodedKey.length == 0) {
279            throw new IllegalArgumentException("Key bytes must not be null or empty");
280        }
281
282        X509EncodedKeySpec spec = new X509EncodedKeySpec(encodedKey);
283        KeyFactory kf = KeyFactory.getInstance(RSA);
284        return kf.generatePublic(spec);
285    }

287    private static byte[] readBytesFromFile(String filename) throws IOException {
288        if (filename == null) {
289            throw new IllegalArgumentException("Filename must not be null");
290        }
291
292        File f = new File(filename);
293        DataInputStream dis = new DataInputStream(new FileInputStream(f));
294        byte[] bytes = new byte[(int) f.length()];
295        dis.readFully(bytes);
296        dis.close();
297        return bytes;
298    }

300    private byte[] hash(byte[] toHash) {
301        try {
302            MessageDigest md = MessageDigest.getInstance(hashMethod);
303            md.update(toHash);
304            return md.digest();
305        } catch (Exception e) {
306            throw new KafkaException(e);
307        }
308    }
309
310    private static byte[] crypt(Cipher c, byte[] plain) throws IllegalBlockSizeException, BadPaddingException {
311        return c.doFinal(plain);
312    }
313
314    private static byte[] crypt(Cipher c, byte[] plain, int offset, int len) throws IllegalBlockSizeException, BadPaddingException {
315        return c.doFinal(plain, offset, len);
316    }

320    public static byte[] concatenate(byte[] a, byte[] b, byte[] c)
321    {
322        if (a != null && b != null && c != null)
323        {
324            byte[] rv = new byte[a.length + b.length + c.length];
325
326            System.arraycopy(a, 0, rv, 0, a.length);
327            System.arraycopy(b, 0, rv, a.length, b.length);
328            System.arraycopy(c, 0, rv, a.length + b.length, c.length);
329
330            return rv;
331        }
332        else if (a == null)
333        {
334            return concatenate(b, c);
335        }
336        else if (b == null)
337        {
338            return concatenate(a, c);
339        }
340        else
341        {
342            return concatenate(a, b);
343        }
344    }
345
346    private static byte[] concatenate(byte[] a, byte[] b) {
347        if (a != null && b != null) {
348            byte[] concat = new byte[a.length + b.length];
349            System.arraycopy(a, 0, concat, 0, a.length);
350            System.arraycopy(b, 0, concat, a.length, b.length);
351            return concat;
352        } else {
353            throw new IllegalArgumentException("arrays must not be null");
354        }
355    }
